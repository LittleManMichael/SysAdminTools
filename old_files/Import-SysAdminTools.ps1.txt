# Import-SysAdmin.ps1 
# Copies SysAdmin module from a central share to the local module path, then imports it. 

param(
    [Parameter(Mandatory=$false)]
    [string]$SourceModuleRoot = '\\server\admin\SysAdmin Powershell Module\SysAdminTools',

    [Parameter(Mandatory=$false)]
    [string]$ModuleName = 'SysAdminTools'
)

$ErrorActionPreference = 'Stop'
$SkippedOverwrite1 = $false
$SkippedOverwrite2 = $false

$psd1Path = Join-Path $SourceModuleRoot "$ModuleName.psd1"
if (-not (Test-Path $psd1Path)) {
    throw "[SysAdmin] Module manifest not found: $psd1Path"
}

$manifest = Import-PowerShellDataFile -Path $psd1Path
$version = [string]$manifest.ModuleVersion

# ProgramFiles root
$targetRoot = Join-Path $env:ProgramFiles "WindowsPowerShell\Modules\$ModuleName\$version"
$targetRootModule = Join-Path $env:ProgramFiles "WindowsPowerShell\Modules\$ModuleName"

# System32 root
$targetRoot2 = Join-Path $env:SystemRoot "System32\WindowsPowershell\v1.0\Modules\$ModuleName\$version"
$targetRoot2Module = Join-Path $env:SystemRoot "System32\WindowsPowershell\v1.0\Modules\$ModuleName"

# Install (copy) if missing to ProgramFiles root
if (Test-Path $targetRoot) {
    Write-Warning "[SysAdmin] $ModuleName already exists at $targetRoot"
    $choice = Read-Host "Overwrite? (Y/N)"

    if ($choice -match '^(Y|y)$') {
        Remove-Item -Path $targetRoot -Recurse -Force -ErrorAction Stop
        New-Item -Path $targetRoot -ItemType Directory -Force | Out-Null
        Write-Host "[SysAdmin] Overwritten SysAdminTools ($version) at $targetRoot" -f Green
        Write-Host ""
    }
    else { 
        Write-Host "[SysAdmin] Skipped overwrite to: $targetRoot" -f Yellow
        Write-Host ""
        $SkippedOverwrite1 = $true
    }
}
else {
    New-Item -Path $targetRoot -ItemType Directory -Force | Out-Null
    Write-Host "[SysAdmin] Created SysAdminTool ($version) at $targetRoot" -f Green
}

# Install (copy) if missing to System32 root
if (Test-Path $targetRoot2) {
    Write-Warning "[SysAdmin] $ModuleName already exists at $targetRoot2"
    $choice = Read-Host "Overwrite? (Y/N)"

    if ($choice -match '^(Y|y)$') {
        Remove-Item -Path $targetRoot2 -Recurse -Force -ErrorAction Stop
        New-Item -Path $targetRoot2 -ItemType Directory -Force | Out-Null
        Write-Host "[SysAdmin] Overwritten SysAdminTools ($version) at $targetRoot2" -f Green
        Write-Host ""
    }
    else { 
        Write-Host "[SysAdmin] Skipped overwrite to: $targetRoot2" -f Yellow
        Write-Host ""
        $SkippedOverwrite2 = $true
    }
}
else {
    New-Item -Path $targetRoot2 -ItemType Directory -Force | Out-Null
    Write-Host "[SysAdmin] Created SysAdminTool ($version) at $targetRoot2" -f Green
}

if (($SkippedOverwrite1 -and $SkippedOverwrite2) -eq $true) {
    Write-Host "[SysAdmin] Overwrite has been skipped. Importing existing module." -f Yellow
    Import-Module $ModuleName
    return
}

# Copy fresh
Copy-Item -Path (Join-Path $SourceModuleRoot '*') -Destination $targetRoot -Recurse -Force -ErrorAction Stop
Copy-Item -Path (Join-Path $SourceModuleRoot '*') -Destination $targetRoot2 -Recurse -Force -ErrorAction Stop

# Import
Import-Module $ModuleName -Force -ErrorAction Stop
Write-Host "[SysAdmin] Imported $ModuleName ($version) Successully" -f Green
