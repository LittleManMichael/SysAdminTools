Function Get-ButtonSize {
    [CmdletBinding()]
    Param(
        [ValidateSet('Small','Medium','Large')] [string]$Size
    )

    switch ($size) {
        'Small'  { return $ButtonSizeSmall }
        'Medium' { return $ButtonSizeMedium }
        'Large'  { return $ButtonSizeLarge }
    }
}

Function Get-CenterPosition {
    [CmdletBinding()]
    Param( 
        [Parameter(Mandatory)] $ParentControl,
        [Parameter(Mandatory)] [int]$ControlWidth,
        [Parameter(Mandatory)] [int]$ControlHeight
    )

    $x = ($ParentControl.ClientSize.Width - $ControlWidth) / 2
    $y = ($ParentControl.ClientSize.Height - $ControlHeight) / 2

    return @{ X = [int]$x ; Y = [int]$y }
}

Function Get-BelowPosition {
    [CmdletBinding()]
    Param(
        [Parameter(Mandatory)]$AboveControl,
        [int]$Spacing = $StandardPadding
    )

    return ($AboveControl.Location.Y + $AboveControl.Height + $Spacing)
}

#region FORM LIFECYCLE MANAGEMENT (Clean Disposal)

Function Invoke-Form {
    <# 
        This shows a form with automatic cleanup to prevent ANY powershell session (including ISE)
        from freezing, or not responding.
        It creates a clean disposal process whether using a defined "Close" button, the red X, or
        by pressing ALT + F4. 
        This will be used to initial WinForms. 
    #>

    [CmdletBinding()]
    Param(
        [Parameter(Mandatory)] [System.Windows.Forms.Form]$Form,
        [switch]$AddCloseButton
    )

    # Add cleanup event handlers
    $Form.Add_FormClosed({
        [System.GC]::Collect()
        [System.GC]::WaitForPendingFinalizers()
    })

    # Optionall add a close button
    if ($AddCloseButton) {
        $btnClose = New-Object System.Windows.Forms.Button 
        $btnClose.Size = New-Object System.Drawing.Size(100, 35)
        $btnClose.Location = New-Object System.Drawing.Point( 
            ($Form.ClientSize.Width -120 ),
            ($Form.ClientSize.Height - 50)
        )
        $btnClose.Text = "Close"
        $btnClose.Font = $FontNormal
        $btnClose.BackColor = $ColorSecondary
        $btnClose.ForeColor = [System.Drawing.Color]::White 
        $btnClose.Anchor = 'Bottom,Right'
        $btnClose.Add_Click({ $Form.Close() })
        
        $Form.Controls.Add($btnClose)
    }

    # Show form with proper cleanup
    try {
        $result = $Form.ShowDialog()
        return $result 
    }
    finally {
        if ($Form -ne $null) { $Form.Dispose() }
        [System.GC]::Collect()
        [System.GC]::WaitForPendingFinalizers()
    }
}

Function Add-CloseButton {
    <# Adds a properly configured close button to a form. Use this when you want manual contorl over the close button placement #>
    [CmdletBinding()]
    Param(
        [Parameter(Mandatory)]$Form,
        [int]$xsize = 100,
        [int]$ysize = 35,
        [Parameter(Mandatory)] [int]$xpos,
        [Parameter(Mandatory)] [int]$ypos,
        [string]$text = "Close"
    )

    $btnClose = Add-Button $xsize $ysize $xpos $ypos $text -Style Secondary
    $btnClose.Add_Click({ $Form.Close() })

    return $btnClose
}

Function Add-Button {
    [CmdletBinding(DefaultParameterSetName='Manual')]
    Param(
        [Parameter(ParameterSetName='Manual', Position=0)] [int]$xsize,
        [Parameter(ParameterSetName='Manual', Position=1)] [int]$ysize,
        [Parameter(ParameterSetName='Preset', Position=0)]
        [Parameter(ParameterSetName='Manual', Position=2)] [int]$xpos,
        [Parameter(ParameterSetName='Preset', Position=1)]
        [Parameter(ParameterSetName='Manual', Position=3)] [int]$ypos,
        [Parameter(ParameterSetName='Preset', Position=2)]
        [Parameter(ParameterSetName='Manual', Position=4)] [string]$text = "",
        [Parameter(ParameterSetName='Preset')]
        [ValidateSet('Small', 'Medium', 'Large')] [string]$Size = 'Medium',
        [ValidateSet('Default', 'Primary', 'Success', 'Danger', 'Warning', 'Info', 'Secondary')] [string]$Style = 'Default',
        [bool]$Enabled = $true
    )

    $button = New-Object System.Windows.Forms.Button

    # Handle sizing
    if ($PSCmdlet.ParameterSetName -eq 'Preset') {
        $btnSize = Get-ButtonSize -Size $Size 
        $button.Size = New-Object System.Drawing.Size($btnSize.Width, $btnSize.Height)
    }
    else {
        $button.Size = New-Object System.Drawing.Size($xsize, $ysize)
    }

    $button.Location = New-Object System.Drawing.Point($xpos, $ypos)
    $button.Text = $text
    $button.Font = $FontNormal 
    $button.Enabled = $Enabled

    # Apply Style Colors
    switch ($Style) {
        'Primary' {
            $button.BackColor = $ColorPrimary 
            $button.ForeColor = [System.Drawing.Color]::White 
        }

        'Success' {
            $button.BackColor = $ColorSuccess 
            $button.ForeColor = [System.Drawing.Color]::White
        }

        'Danger' {
            $button.BackColor = $ColorDanger
            $button.ForeColor = [System.Drawing.Color]::White 
        }

        'Warning' {
            $button.BackColor = $ColorWarning 
            $button.ForeColor = [System.Drawing.Color]::Black
        }
        'Info' {
            $button.BackColor = $ColorInfo 
            $button.ForeColor = [System.Drawing.Color]::White 
        }

        'Secondary' {
            $button.BackColor = $ColorSecondary
            $button.ForeColor = [System.Drawing.Color]::White
        }
    }

    return $button
}

Function Add-Label {
    [CmdletBinding()]
    Param(
        [Parameter(Mandatory)] [int]$xsize,
        [Parameter(Mandatory)] [int]$ysize,
        [Parameter(Mandatory)] [int]$xpos,
        [Parameter(Mandatory)] [int]$ypos,
        [string]$text = "",
        [ValidateSet('Small', 'Normal', 'Large', 'Bold')] [string]$FontSize = 'Normal',
        [ValidateSet('Left', 'Center', 'Right')] [string]$TextAlign = 'Center',
        [switch]$NoBorder,
        [System.Drawing.Color]$ForeColor = $ColorTextPrimary ,
        [System.Drawing.Color]$BackColor = $ColorControlBackground 
    )

    $label = New-Object System.Windows.Forms.Label
    $label.Size = New-Object System.Drawing.Size($xsize, $ysize)
    $label.Location = New-Object System.Drawing.Point($xpos, $ypos)
    $label.Text = $text 
    $label.BackColor = $BackColor
    $label.ForeColor = $ForeColor

    # Set font based on size
    switch ($FontSize) {
        'Small'  { $label.Font = $FontSmall }
        'Normal' { $label.Font = $FontNormal }
        'Large'  { $label.Font = $FontLarge }
        'Bold'   { $label.Font = $FontBold }
    }

    # Set text alignment
    switch ($TextAlign) {
        'Left'   { $label.TextAlign = 'MiddleLeft' }
        'Center' { $label.TextAlign = 'MiddleCenter' }
        'Right'  { $label.TextAlign = 'MiddleRight' }
    }

    # set border
    if (-not $NoBorder) {
        $label.BorderStyle = 'FixedSingle'
    }

    return $label
}

Function Add-OpenFileDialog {
    [CmdletBinding()]
    Param(
        [string]$Filter = "All Files (*.*)|*.*",
        [string]$Title = "Select File",
        [string]$InitialDirectory = ""
    )

    $openFileDialog = New-Object System.Windows.Forms.OpenFileDialog
    $openFileDialog.Filter = $Filter
    $openFileDialog.Title = $Title

    if ($InitialDirectory) {
        $openFileDialog.InitialDirectory = $InitialDirectory
    }

    $result = $openFileDialog.ShowDialog()

    if ($result -eq 'OK') {
        return $openFileDialog.FileName
    }
    else {
        return $null
    }
}
